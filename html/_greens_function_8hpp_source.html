127inline ComplexType GreensFunction::operator()(ComplexType z) const {
128 if(Vanishing)
129 return 0;
130 else {
131 return std::accumulate(parts.begin(),
132 parts.end(),
133 ComplexType(0),
134 [z](ComplexType s, GreensFunctionPart const& p) { return s + p(z); });
135 }
136}
