472 inline ComplexType
473 TwoParticleGFPart::NonResonantTerm::operator()(ComplexType z1, ComplexType z2, ComplexType z3) const {
474  return isz4 ? Coeff / ((z1 - Poles[0]) * (z1 + z2 + z3 - Poles[0] - Poles[1] - Poles[2]) * (z3 - Poles[2])) :
475  Coeff / ((z1 - Poles[0]) * (z2 - Poles[1]) * (z3 - Poles[2]));
476 }

478 inline ComplexType TwoParticleGFPart::ResonantTerm::operator()(ComplexType z1,
479  ComplexType z2,
480  ComplexType z3,
481  RealType DeltaTolerance) const {
482  ComplexType Diff;
483  if(isz1z2) {
484  Diff = z1 + z2 - Poles[0] - Poles[1];
485  return (std::abs(Diff) < DeltaTolerance ? ResCoeff : (NonResCoeff / Diff)) /
486  ((z1 - Poles[0]) * (z3 - Poles[2]));
487  } else {
488  Diff = z2 + z3 - Poles[1] - Poles[2];
489  return (std::abs(Diff) < DeltaTolerance ? ResCoeff : (NonResCoeff / Diff)) /
490  ((z1 - Poles[0]) * (z3 - Poles[2]));
491  }
