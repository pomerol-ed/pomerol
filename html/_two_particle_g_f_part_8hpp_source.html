437 inline ComplexType
438 TwoParticleGFPart::NonResonantTerm::operator()(ComplexType z1, ComplexType z2, ComplexType z3) const {
439  return isz4 ? Coeff / ((z1 - Poles[0]) * (z1 + z2 + z3 - Poles[0] - Poles[1] - Poles[2]) * (z3 - Poles[2])) :
440  Coeff / ((z1 - Poles[0]) * (z2 - Poles[1]) * (z3 - Poles[2]));
441 }

443 inline ComplexType TwoParticleGFPart::ResonantTerm::operator()(ComplexType z1,
444  ComplexType z2,
445  ComplexType z3,
446  RealType DeltaTolerance) const {
447  ComplexType Diff;
448  if(isz1z2) {
449  Diff = z1 + z2 - Poles[0] - Poles[1];
450  return (std::abs(Diff) < DeltaTolerance ? ResCoeff : (NonResCoeff / Diff)) /
451  ((z1 - Poles[0]) * (z3 - Poles[2]));
452  } else {
453  Diff = z2 + z3 - Poles[1] - Poles[2];
454  return (std::abs(Diff) < DeltaTolerance ? ResCoeff : (NonResCoeff / Diff)) /
455  ((z1 - Poles[0]) * (z3 - Poles[2]));
456  }
457 }
