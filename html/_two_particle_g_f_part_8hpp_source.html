475inline ComplexType
476TwoParticleGFPart::NonResonantTerm::operator()(ComplexType z1, ComplexType z2, ComplexType z3) const {
477 return isz4 ? Coeff / ((z1 - Poles[0]) * (z1 + z2 + z3 - Poles[0] - Poles[1] - Poles[2]) * (z3 - Poles[2])) :
478 Coeff / ((z1 - Poles[0]) * (z2 - Poles[1]) * (z3 - Poles[2]));
479}
480
481inline ComplexType TwoParticleGFPart::ResonantTerm::operator()(ComplexType z1,
482 ComplexType z2,
483 ComplexType z3,
484 RealType DeltaTolerance) const {
485 ComplexType Diff;
486 if(isz1z2) {
487 Diff = z1 + z2 - Poles[0] - Poles[1];
488 return (std::abs(Diff) <= DeltaTolerance ? ResCoeff : (NonResCoeff / Diff)) /
489 ((z1 - Poles[0]) * (z3 - Poles[2]));
490 } else {
491 Diff = z2 + z3 - Poles[1] - Poles[2];
492 return (std::abs(Diff) <= DeltaTolerance ? ResCoeff : (NonResCoeff / Diff)) /
493 ((z1 - Poles[0]) * (z3 - Poles[2]));
494 }
495}
