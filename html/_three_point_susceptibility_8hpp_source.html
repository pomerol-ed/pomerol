162inline ComplexType ThreePointSusceptibility::operator()(ComplexType z1, ComplexType z2) const {
163 if(Vanishing)
164 return 0;
165 else {
166 return std::accumulate(
167 parts.begin(),
168 parts.end(),
169 ComplexType(0),
170 [z1, z2](ComplexType s, ThreePointSusceptibilityPart const& p) { return s + p(z1, z2); });
171 }
172}
