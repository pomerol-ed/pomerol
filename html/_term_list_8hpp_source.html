41template <typename TermType> class TermList {
42
44 using Hash = typename TermType::Hash;
46 using KeyEqual = typename TermType::KeyEqual;
48 using IsNegligible = typename TermType::IsNegligible;
49
51 std::unordered_set<TermType, Hash, KeyEqual> data;
53 IsNegligible is_negligible;
54
55public:
60 TermList(Hash const& hasher, KeyEqual const& key_equal, IsNegligible const& is_negligible)
61 : data(1, hasher, key_equal), is_negligible(is_negligible) {}
62
65 void add_term(TermType const& term) {
66 auto it = data.find(term);
67 if(it == data.end()) { // new term
68 data.emplace(term);
69 } else { // similar term
70 TermType sum = *it;
71 sum += term;
72 data.erase(*it);
73 if(!is_negligible(sum, data.size() + 1))
74 data.emplace(sum);
75 }
76 }

94 template <typename... Args> ComplexType operator()(Args&&... args) const {
95 ComplexType res = 0;
96 for(auto const& t : data) {
97 // cppcheck-suppress useStlAlgorithm
98 res += t(args...);
99 }
100 return res;
101 }

106 void broadcast(MPI_Comm const& comm, int root) {
107 auto hasher = data.hash_function();
108 auto key_eq = data.key_eq();
109
110 hasher.broadcast(comm, root);
111 key_eq.broadcast(comm, root);
112
113 long n_terms;
114 if(pMPI::rank(comm) == root) { // Broadcast the terms from this process
115 n_terms = data.size();
116 std::vector<TermType> v(data.begin(), data.end());
117 MPI_Bcast(&n_terms, 1, MPI_LONG, root, comm);
118 MPI_Bcast(v.data(), v.size(), TermType::mpi_datatype(), root, comm);
119 } else { // Receive terms
120 MPI_Bcast(&n_terms, 1, MPI_LONG, root, comm);
121 std::vector<TermType> v(n_terms);
122 MPI_Bcast(v.data(), v.size(), TermType::mpi_datatype(), root, comm);
123 data = std::unordered_set<TermType, Hash, KeyEqual>(v.begin(), v.end(), 1, hasher, key_eq);
124 }
125
126 is_negligible.broadcast(comm, root);
127 }
128
130 bool check_terms() const {
131 return std::none_of(data.begin(), data.end(), [this](TermType const& t) {
132 return is_negligible(t, data.size() + 1);
133 });
134 }
135};
