62  void add_term(TermType const& term) {
63  auto it = data.find(term);
64  if(it == data.end()) { // new term
65  data.emplace(term);
66  } else { // similar term
67  TermType sum = *it;

101  void broadcast(MPI_Comm const& comm, int root) {

111  } else { // Receive terms

122  bool check_terms() const {
123  if(size() == 0)
124  return true;
125  auto prev_it = data.begin();
126  if(is_negligible(*prev_it, data.size() + 1))
127  return false;
128  if(size() == 1)
129  return true;
130 
131  Compare const& compare = data.key_comp();
132 
133  auto it = prev_it;
134  for(++it; it != data.end(); ++it, ++prev_it) {
135  if(is_negligible(*it, data.size() + 1) || !compare(*prev_it, *it))
136  return false;
137  }
138  return true;
139  }
