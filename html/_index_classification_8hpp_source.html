41 template <typename... IndexTypes> class IndexClassification {
42 public:
44  using IndexInfo = std::tuple<IndexTypes...>;
45 
46 private:
48  std::map<IndexInfo, ParticleIndex> InfoToIndices;
50  std::vector<IndexInfo> IndicesToInfo;

58  template <typename ScalarType>
59  explicit IndexClassification(Operators::expression<ScalarType, IndexTypes...> const& H) {
60  // Collect indices of fermionic operators in the Hamiltonian
61  for(auto const& mon : H) {
62  for(auto const& g : mon.monomial) {
63  if(libcommute::is_fermion(g))
64  InfoToIndices.emplace(g.indices(), 0);
65  }
66  }
67  UpdateMaps();
68  }
70  IndexClassification() = default;

77  }

92  ParticleIndex getIndex(IndexInfo const& info) const {
93  auto it = InfoToIndices.find(info);
94  if(it != InfoToIndices.end())
95  return it->second;
96  else {
97  std::stringstream ss;
98  libcommute::print_tuple(ss, info);
99  throw std::runtime_error("Wrong indices " + ss.str());
100  }
101  }

110  IndexInfo const& getInfo(ParticleIndex in) const {
111  if(in >= InfoToIndices.size())
112  throw std::runtime_error("Wrong particle index " + std::to_string(in));
113  return IndicesToInfo[in];

123  friend std::ostream& operator<<(std::ostream& os, IndexClassification const& ic) {

132 private:
134  void UpdateMaps() {
135  IndicesToInfo.clear();
136  IndicesToInfo.reserve(InfoToIndices.size());
137  for(auto& ii : InfoToIndices) {
138  ii.second = IndicesToInfo.size();
139  IndicesToInfo.emplace_back(ii.first);
140  }
141  }
