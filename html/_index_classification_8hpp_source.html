40template <typename... IndexTypes> class IndexClassification {
41public:
43 using IndexInfo = std::tuple<IndexTypes...>;
44
45private:
47 std::map<IndexInfo, ParticleIndex> InfoToIndices;
49 std::vector<IndexInfo> IndicesToInfo;
50
51public:
57 template <typename ScalarType>
58 explicit IndexClassification(Operators::expression<ScalarType, IndexTypes...> const& H) {
59 // Collect indices of fermionic operators in the Hamiltonian
60 for(auto const& mon : H) {
61 for(auto const& g : mon.monomial) {
62 if(libcommute::is_fermion(g))
63 InfoToIndices.emplace(g.indices(), 0);
64 }
65 }
66 UpdateMaps();
67 }
69 IndexClassification() = default;
70
73 void addInfo(IndexInfo const& info) {
74 InfoToIndices.emplace(info, 0);
75 UpdateMaps();
76 }
77
80 void addInfo(IndexTypes... indices) {
81 InfoToIndices.emplace(IndexInfo{indices...}, 0);
82 UpdateMaps();
83 }
84
87 bool checkIndex(ParticleIndex in) const { return in < InfoToIndices.size(); }
88
91 ParticleIndex getIndex(IndexInfo const& info) const {
92 auto it = InfoToIndices.find(info);
93 if(it != InfoToIndices.end())
94 return it->second;
95 else {
96 throw std::runtime_error("Wrong indices " + libcommute::tuple_to_string(info));
97 }
98 }
99
103 ParticleIndex getIndex(IndexTypes... indices) const { return getIndex(std::make_tuple(indices...)); }
104
107 IndexInfo const& getInfo(ParticleIndex in) const {
108 if(in >= InfoToIndices.size())
109 throw std::runtime_error("Wrong particle index " + std::to_string(in));
110 return IndicesToInfo[in];
111 }
112
114 ParticleIndex getIndexSize() const { return InfoToIndices.size(); }
115
120 friend std::ostream& operator<<(std::ostream& os, IndexClassification const& ic) {
121 for(ParticleIndex i = 0; i < ic.InfoToIndices.size(); ++i) {
122 os << "Index " << i << " = (";
123 os << libcommute::tuple_to_string(ic.IndicesToInfo[i]);
124 os << ")\n";
125 }
126 return os;
127 }
128
129private:
131 void UpdateMaps() {
132 IndicesToInfo.clear();
133 IndicesToInfo.reserve(InfoToIndices.size());
134 for(auto& ii : InfoToIndices) {
135 ii.second = IndicesToInfo.size();
136 IndicesToInfo.emplace_back(ii.first);
137 }
138 }
139};
