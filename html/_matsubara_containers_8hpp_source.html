80template <typename SourceObject> inline void MatsubaraContainer4<SourceObject>::fill(long NumberOfMatsubaras) {
81 this->NumberOfMatsubaras = NumberOfMatsubaras;
82
83 if(NumberOfMatsubaras == 0) {
84 Values.clear();
85 FermionicIndexOffset.clear();
86 return;
87 } else {
88 Values.resize(4 * NumberOfMatsubaras - 1);
89 FermionicIndexOffset.resize(4 * NumberOfMatsubaras - 1);
90 }
91
92 // \omega_1 = \nu, \omega_3 = \nu', \omega_1+\omega_2 = \Omega
93 for(long BosonicIndexV = 0; BosonicIndexV <= 4 * NumberOfMatsubaras - 2; ++BosonicIndexV) {
94 long BosonicIndex = BosonicIndexV - 2 * NumberOfMatsubaras;
95 long FermionicMatrixSize = 2 * NumberOfMatsubaras - std::abs(BosonicIndex + 1);
96 Values[BosonicIndexV].resize(FermionicMatrixSize, FermionicMatrixSize);
97 FermionicIndexOffset[BosonicIndexV] = (BosonicIndex < 0 ? 0 : BosonicIndex + 1) - NumberOfMatsubaras;
98
99 for(long NuIndexM = 0; NuIndexM < FermionicMatrixSize; ++NuIndexM)
100 for(long NupIndexM = 0; NupIndexM < FermionicMatrixSize; ++NupIndexM) {
101 long MatsubaraNumber1 = NuIndexM + FermionicIndexOffset[BosonicIndexV];
102 long MatsubaraNumber2 = BosonicIndex - MatsubaraNumber1;
103 long MatsubaraNumber3 = NupIndexM + FermionicIndexOffset[BosonicIndexV];
104 Values[BosonicIndexV](NuIndexM, NupIndexM) =
105 Source.value(MatsubaraNumber1, MatsubaraNumber2, MatsubaraNumber3);
106 }
107 }
108}

110template <typename SourceObject>
111inline ComplexType MatsubaraContainer4<SourceObject>::operator()(long MatsubaraNumber1,
112 long MatsubaraNumber2,
113 long MatsubaraNumber3) const {
114 long BosonicIndexV = MatsubaraNumber2 + MatsubaraNumber1 + 2 * NumberOfMatsubaras;
115 if(BosonicIndexV >= 0 && BosonicIndexV <= 2 * (2 * NumberOfMatsubaras - 1)) {
116 long NuIndexM = MatsubaraNumber1 - FermionicIndexOffset[BosonicIndexV];
117 long NupIndexM = MatsubaraNumber3 - FermionicIndexOffset[BosonicIndexV];
118 if(NuIndexM >= 0 && NuIndexM < Values[BosonicIndexV].rows() && NupIndexM >= 0 &&
119 NupIndexM < Values[BosonicIndexV].cols())
120 return Values[BosonicIndexV](NuIndexM, NupIndexM);
121 }
122
123 DEBUG("MatsubaraContainer4 at " << this << ": "
124 << "cache miss for n1 = " << MatsubaraNumber1 << ", "
125 << "n2 = " << MatsubaraNumber2 << ", "
126 << "n3 = " << MatsubaraNumber3 << " (NumberOfMatsubaras = " << NumberOfMatsubaras
127 << "), fetching a raw value from " << &Source);
128
129 return Source.value(MatsubaraNumber1, MatsubaraNumber2, MatsubaraNumber3);
130}
