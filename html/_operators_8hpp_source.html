69template <std::size_t... Ints> struct index_sequence {
70 using type = index_sequence;
71 using value_type = std::size_t;
72 static constexpr std::size_t size() noexcept { return sizeof...(Ints); }
73};

76template <class Sequence1, class Sequence2> struct _merge_and_renumber;
77
78template <std::size_t... I1, std::size_t... I2>
79struct _merge_and_renumber<index_sequence<I1...>, index_sequence<I2...>>
80 : index_sequence<I1..., (sizeof...(I1) + I2)...> {};

84template <std::size_t N>
85struct make_index_sequence
86 : _merge_and_renumber<typename make_index_sequence<N / 2>::type, typename make_index_sequence<N - N / 2>::type> {};
87
88template <> struct make_index_sequence<0> : index_sequence<> {};
89template <> struct make_index_sequence<1> : index_sequence<0> {};

91//
92// Apply a function to a tuple of arguments
93//
94
95template <typename T> make_index_sequence<std::tuple_size<typename std::decay<T>::type>::value> make_seq() {
96 return {};
97}
98
99template <std::size_t N, typename T>
100using element_t = typename std::tuple_element<N, typename std::decay<T>::type>::type;
101
102template <typename F, typename ArgsT, std::size_t... Is>
103auto apply_impl(F&& f, ArgsT&& args, index_sequence<Is...>)
104 -> decltype(f(static_cast<element_t<Is, ArgsT>>(std::get<Is>(std::forward<ArgsT>(args)))...)) {
105 return f(static_cast<element_t<Is, ArgsT>>(std::get<Is>(std::forward<ArgsT>(args)))...);
106}
107
108template <typename F, typename ArgsT>
109auto apply(F&& f, ArgsT&& args)
110 -> decltype(apply_impl(std::forward<F>(f), std::forward<ArgsT>(args), make_seq<ArgsT>())) {
111 return apply_impl(std::forward<F>(f), std::forward<ArgsT>(args), make_seq<ArgsT>());
112}

127template <typename... IndexTypes>
128expression<double, IndexTypes...> N(std::vector<std::tuple<IndexTypes...>> const& Indices) {
129 expression<double, IndexTypes...> res;
130 for(auto const& i : Indices)
131 res += Detail::apply(n<double, IndexTypes...>, i); // cppcheck-suppress useStlAlgorithm
132 return res;
133}
134
145template <typename... IndexTypes>
146expression<double, IndexTypes...> Sz(std::vector<std::tuple<IndexTypes...>> const& SpinUpIndices,
147 std::vector<std::tuple<IndexTypes...>> const& SpinDownIndices) {
148 expression<double, IndexTypes...> res;
149 for(auto const& i : SpinUpIndices)
150 res += 0.5 * Detail::apply(n<double, IndexTypes...>, i); // cppcheck-suppress useStlAlgorithm
151 for(auto const& i : SpinDownIndices)
152 res -= 0.5 * Detail::apply(n<double, IndexTypes...>, i); // cppcheck-suppress useStlAlgorithm
153 return res;
154}
