158  template <typename... IndexTypes> void prepare(HilbertSpace<IndexTypes...> const& HS) {
159  if(getStatus() >= Prepared)
160  return;
161 
162  if(HS.getStatus() != Computed) { // Hilbert space has not been partitioned
163  mapPartsFromRight.emplace(0, 0);
164  mapPartsFromLeft.emplace(0, 0);
165  LeftRightBlocks.insert(BlockMapping(0, 0));
166 
167  if(MOpComplex)
168  parts.emplace_back(getMOp<true>(), S, H.getPart(0), H.getPart(0));
169  else
170  parts.emplace_back(getMOp<false>(), S, H.getPart(0), H.getPart(0));
171 
172  Status = Prepared;
173  return;
174  }
175 
176  auto const& FullHS = HS.getFullHilbertSpace();
177  auto const& Partition = HS.getSpacePartition();
178  auto Connections = MOpComplex ? Partition.find_connections(getMOp<true>(), FullHS) :
179  Partition.find_connections(getMOp<false>(), FullHS);
180 
181  parts.reserve(Connections.size());
182  for(auto const& Conn : Connections) {
183  mapPartsFromRight.emplace(Conn.first, parts.size());
184  mapPartsFromLeft.emplace(Conn.second, parts.size());
185  LeftRightBlocks.insert(BlockMapping(Conn.second, Conn.first));
186 
187  if(MOpComplex)
188  parts.emplace_back(getMOp<true>(), S, H.getPart(Conn.first), H.getPart(Conn.second));
189  else
190  parts.emplace_back(getMOp<false>(), S, H.getPart(Conn.first), H.getPart(Conn.second));
191  }
192 
193  Status = Prepared;
194  }
