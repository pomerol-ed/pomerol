160  template <typename... IndexTypes> void prepare(HilbertSpace<IndexTypes...> const& HS) {
161  if(getStatus() >= Prepared)
162  return;
163 
164  if(HS.getStatus() != Computed) { // Hilbert space has not been partitioned
165  mapPartsFromRight.emplace(0, 0);
166  mapPartsFromLeft.emplace(0, 0);
167  LeftRightBlocks.insert(BlockMapping(0, 0));
168 
169  if(MOpComplex)
170  parts.emplace_back(getMOp<true>(), S, H.getPart(0), H.getPart(0));
171  else
172  parts.emplace_back(getMOp<false>(), S, H.getPart(0), H.getPart(0));
173 
174  Status = Prepared;
175  return;
176  }
177 
178  auto const& FullHS = HS.getFullHilbertSpace();
179  auto const& Partition = HS.getSpacePartition();
180  auto Connections = MOpComplex ? Partition.find_connections(getMOp<true>(), FullHS) :
181  Partition.find_connections(getMOp<false>(), FullHS);
182 
183  parts.reserve(Connections.size());
184  for(auto const& Conn : Connections) {
185  mapPartsFromRight.emplace(Conn.first, parts.size());
186  mapPartsFromLeft.emplace(Conn.second, parts.size());
187  LeftRightBlocks.insert(BlockMapping(Conn.second, Conn.first));
188 
189  if(MOpComplex)
190  parts.emplace_back(getMOp<true>(), S, H.getPart(Conn.first), H.getPart(Conn.second));
191  else
192  parts.emplace_back(getMOp<false>(), S, H.getPart(Conn.first), H.getPart(Conn.second));
193  }
194 
195  Status = Prepared;
196  }
