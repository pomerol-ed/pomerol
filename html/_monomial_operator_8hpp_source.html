160 template <typename... IndexTypes> void prepare(HilbertSpace<IndexTypes...> const& HS) {
161 if(getStatus() >= Prepared)
162 return;
163
164 if(HS.getStatus() != Computed) { // Hilbert space has not been partitioned
165 mapPartsFromRight.emplace(0, 0);
166 mapPartsFromLeft.emplace(0, 0);
167 LeftRightBlocks.insert(BlockMapping(0, 0));
168
169 if(MOpComplex)
170 parts.emplace_back(getMOp<true>(), S, H.getPart(0), H.getPart(0));
171 else
172 parts.emplace_back(getMOp<false>(), S, H.getPart(0), H.getPart(0));
173
174 Status = Prepared;
175 return;
176 }
177
178 auto const& Partition = HS.getSpacePartition();
179 auto Connections =
180 MOpComplex ? Partition.find_connections(getMOp<true>()) : Partition.find_connections(getMOp<false>());
181
182 parts.reserve(Connections.size());
183 for(auto const& Conn : Connections) {
184 mapPartsFromRight.emplace(Conn.first, parts.size());
185 mapPartsFromLeft.emplace(Conn.second, parts.size());
186 LeftRightBlocks.insert(BlockMapping(Conn.second, Conn.first));
187
188 if(MOpComplex)
189 parts.emplace_back(getMOp<true>(), S, H.getPart(Conn.first), H.getPart(Conn.second));
190 else
191 parts.emplace_back(getMOp<false>(), S, H.getPart(Conn.first), H.getPart(Conn.second));
192 }
193
194 Status = Prepared;
195 }

202 void compute(RealType Tolerance = 1e-8, MPI_Comm const& comm = MPI_COMM_WORLD);
203
204private:
205 // Implementation details
206 void checkPrepared() const;

210class FieldOperator : public MonomialOperator {
212 ParticleIndex Index;
213
214public:
224 template <typename ScalarType, typename... IndexTypes>
225 FieldOperator(libcommute::expression<ScalarType, IndexTypes...> const& FO,
226 HilbertSpace<IndexTypes...> const& HS,
227 StatesClassification const& S,
228 Hamiltonian const& H,
229 ParticleIndex Index)
230 : MonomialOperator(FO, HS, S, H), Index(Index) {
231 auto const& mon = FO.get_monomials().cbegin()->first;
232 if(mon.size() != 1 || !is_fermion(mon[0]))
233 throw std::runtime_error("Expected a single-fermion monomial expression");
234 }
235
