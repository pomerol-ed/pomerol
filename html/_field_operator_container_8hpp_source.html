55 template <typename... IndexTypes>
56 FieldOperatorContainer(IndexClassification<IndexTypes...> const& IndexInfo,
57 HilbertSpace<IndexTypes...> const& HS,
58 StatesClassification const& S,
59 Hamiltonian const& H,
60 std::set<ParticleIndex> in = {}) {
61 if(in.empty()) {
62 for(ParticleIndex p = 0; p < IndexInfo.getIndexSize(); ++p) {
63 in.insert(p);
64 }
65 }
66 for(auto p : in) {
67 mapCreationOperators.emplace(p, CreationOperator(IndexInfo, HS, S, H, p));
68 mapAnnihilationOperators.emplace(p, AnnihilationOperator(IndexInfo, HS, S, H, p));
69 }
70 }
71
75 template <typename... IndexTypes> void prepareAll(HilbertSpace<IndexTypes...> const& HS) {
76 for(auto& CX : mapCreationOperators)
77 CX.second.prepare(HS);
78 for(auto& C : mapAnnihilationOperators)
79 C.second.prepare(HS);
80 }

86 void computeAll(RealType Tolerance = 1e-8);
