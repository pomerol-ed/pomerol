82  inline std::unique_ptr<libcommute::elementary_space<IndexTypes...>>
83  operator()(libcommute::generator<IndexTypes...> const& g) const {
84  using namespace libcommute;
85  if(is_fermion(g)) {
86  return make_unique<elementary_space_fermion<IndexTypes...>>(g.indices());
87  } else if(is_boson(g)) {
88  auto it = bits_per_boson.find(g.indices());
89  unsigned int bits = (it == bits_per_boson.end()) ? 1 : it->second;
90  return make_unique<elementary_space_boson<IndexTypes...>>(bits, g.indices());
91  } else {
92  std::stringstream ss;
93  ss << "Unexpected algebra generator " << g;
94  throw std::runtime_error(ss.str());
95  }
96  }
97  };

141  void compute() {
142  if(getStatus() >= Computed)
143  return;
144 
145  // Phase I of auto-partition algorithm
146  if(HamiltonianComplex) {
147  auto const& op = *std::static_pointer_cast<LOperatorTypeRC<true>>(HOp);
148  Partition.reset(new libcommute::space_partition(op, FullHilbertSpace));
149  } else {
150  auto const& op = *std::static_pointer_cast<LOperatorTypeRC<false>>(HOp);
151  Partition.reset(new libcommute::space_partition(op, FullHilbertSpace));
152  }
153  // Phase II of auto-partition algorithm
154  for(ParticleIndex in = 0; in < IndexInfo.getIndexSize(); ++in) {
155  auto const& info = IndexInfo.getInfo(in);
156  using Operators::c_dag;
157  using Operators::c;
158  auto Cd =
159  LOperatorType<RealType>(Operators::Detail::apply(c_dag<double, IndexTypes...>, info), FullHilbertSpace);
160  auto C =
161  LOperatorType<RealType>(Operators::Detail::apply(c<double, IndexTypes...>, info), FullHilbertSpace);
162  Partition->merge_subspaces(Cd, C, FullHilbertSpace, false);
163  }
164 
165  setStatus(Computed);
166  }
167
