76 struct boson_es_constructor_from_map {
77 std::map<std::tuple<IndexTypes...>, unsigned int> const& bits_per_boson;
78
79 explicit boson_es_constructor_from_map(std::map<std::tuple<IndexTypes...>, unsigned int> const& bits_per_boson)
80 : bits_per_boson(bits_per_boson) {}
81
82 inline std::unique_ptr<libcommute::elementary_space<IndexTypes...>>
83 operator()(libcommute::generator<IndexTypes...> const& g) const {
84 using namespace libcommute;
85 if(is_fermion(g)) {
86 return make_unique<elementary_space_fermion<IndexTypes...>>(g.indices());
87 } else if(is_boson(g)) {
88 auto it = bits_per_boson.find(g.indices());
89 unsigned int bits = (it == bits_per_boson.end()) ? 1 : it->second;
90 return make_unique<elementary_space_boson<IndexTypes...>>(1 << bits, g.indices());
91 } else {
92 std::stringstream ss;
93 ss << "Unexpected algebra generator " << g;
94 throw std::runtime_error(ss.str());
95 }
96 }
97 };

144 void compute() {
145 if(getStatus() >= Computed)
146 return;
147
148 // Phase I of auto-partition algorithm
149 if(HamiltonianComplex) {
150 auto const& op = *std::static_pointer_cast<LOperatorTypeRC<true>>(HOp);
151 Partition.reset(new SpacePartitionType(op, FullHilbertSpace));
152 } else {
153 auto const& op = *std::static_pointer_cast<LOperatorTypeRC<false>>(HOp);
154 Partition.reset(new SpacePartitionType(op, FullHilbertSpace));
155 }
156 // Phase II of auto-partition algorithm
157 for(ParticleIndex in = 0; in < IndexInfo.getIndexSize(); ++in) {
158 auto const& info = IndexInfo.getInfo(in);
159 using Operators::c_dag;
160 using Operators::c;
161 auto Cd =
162 LOperatorType<RealType>(Operators::Detail::apply(c_dag<double, IndexTypes...>, info), FullHilbertSpace);
163 auto C =
164 LOperatorType<RealType>(Operators::Detail::apply(c<double, IndexTypes...>, info), FullHilbertSpace);
165 Partition->merge_subspaces(Cd, C, false);
166 }
167
168 setStatus(Computed);
169 }
170
172 IndexClassification<IndexTypes...> const& getIndexInfo() const { return IndexInfo; }
173
175 FullHilbertSpaceType const& getFullHilbertSpace() const { return FullHilbertSpace; }
176
179 SpacePartitionType const& getSpacePartition() const {
180 if(getStatus() < Computed)
181 throw std::runtime_error("Hilbert space partition has not been computed");
182 return *Partition;
183 }
184};
