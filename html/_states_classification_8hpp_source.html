59 template <typename... IndexTypes> void compute(HilbertSpace<IndexTypes...> const& HS) {
60 if(getStatus() == Computed)
61 return;
62 auto const& FullHilbertSpace = HS.getFullHilbertSpace();
63 auto Dim = FullHilbertSpace.dim();
64 if(HS.getStatus() == Computed) { // Multiple blocks revealed by HS
65 initMultipleBlocks(HS.getSpacePartition());
66 } else { // Just one block
67 initSingleBlock(Dim);
68 }
69 setStatus(Computed);
70 }

107 void initSingleBlock(QuantumState Dim);
110 template <typename SpacePartitionType> void initMultipleBlocks(SpacePartitionType const& partition) {
111 StateBlockIndex.resize(partition.dim());
112 StatesContainer.resize(partition.n_subspaces(), std::vector<QuantumState>());
113 foreach(partition, [this](QuantumState State, BlockNumber Block) {
114 StateBlockIndex[State] = Block;
115 StatesContainer[Block].push_back(State);
116 })
117 ;
118 }
