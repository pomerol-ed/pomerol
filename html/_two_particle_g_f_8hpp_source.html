169 inline ComplexType TwoParticleGF::operator()(ComplexType z1, ComplexType z2, ComplexType z3) const {
170  if(Vanishing)
171  return 0;
172  else {
173  return std::accumulate(parts.begin(),
174  parts.end(),
175  ComplexType(0),
176  [z1, z2, z3](ComplexType s, TwoParticleGFPart const& p) { return s + p(z1, z2, z3); });
