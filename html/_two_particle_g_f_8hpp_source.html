174 inline ComplexType TwoParticleGF::operator()(ComplexType z1, ComplexType z2, ComplexType z3) const {
175  if(Vanishing)
176  return 0;
177  else {
178  return std::accumulate(parts.begin(),
179  parts.end(),
180  ComplexType(0),
181  [z1, z2, z3](ComplexType s, TwoParticleGFPart const& p) { return s + p(z1, z2, z3); });
