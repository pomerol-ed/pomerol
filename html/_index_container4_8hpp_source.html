137template <typename ElementType>
138inline ComplexType ElementWithPermFreq<ElementType>::operator()(long MatsubaraNumber1,
139 long MatsubaraNumber2,
140 long MatsubaraNumber3) const {
141 long MatsubaraNumbers[4] = {MatsubaraNumber1,
142 MatsubaraNumber2,
143 MatsubaraNumber3,
144 MatsubaraNumber1 + MatsubaraNumber2 - MatsubaraNumber3};
145 return (*pElement)(MatsubaraNumbers[FrequenciesPermutation.perm[0]],
146 MatsubaraNumbers[FrequenciesPermutation.perm[1]],
147 MatsubaraNumbers[FrequenciesPermutation.perm[2]]) *
148 RealType(FrequenciesPermutation.sign);
149}

190template <typename ElementType, typename SourceObject>
191inline ElementWithPermFreq<ElementType>&
192IndexContainer4<ElementType, SourceObject>::create(IndexCombination4 const& Indices) {
193 std::shared_ptr<ElementType> pElement(Source.createElement(Indices));
194 auto iter = ElementsMap.emplace(Indices, ElementWithPermFreq<ElementType>(pElement, permutations4[0])).first;
195
196 DEBUG("IndexContainer4::fill() at " << this << ": "
197 << "added an element with indices " << Indices << " and frequency permutation "
198 << permutations4[0] << " (" << pElement << ").");
199
200 bool SameCIndices = (Indices.Index1 == Indices.Index2);
201 bool SameCXIndices = (Indices.Index3 == Indices.Index4);
202
203 NonTrivialElements.emplace(Indices, pElement);
204
205 if(!SameCIndices) {
206 IndexCombination4 Indices2134(Indices.Index2, Indices.Index1, Indices.Index3, Indices.Index4);
207 if(!isInContainer(Indices2134)) {
208 ElementsMap.emplace(Indices2134, ElementWithPermFreq<ElementType>(pElement, permutations4[6]));
209 DEBUG("IndexContainer4::fill() at " << this << ": "
210 << "added an element with indices " << Indices
211 << " and frequency permutation " << permutations4[6] << " (" << pElement
212 << ").");
213 }
214 }
215 if(!SameCXIndices) {
216 IndexCombination4 Indices1243(Indices.Index1, Indices.Index2, Indices.Index4, Indices.Index3);
217 if(!isInContainer(Indices1243)) {
218 ElementsMap.emplace(Indices1243, ElementWithPermFreq<ElementType>(pElement, permutations4[1]));
219 DEBUG("IndexContainer4::fill() at " << this << ": "
220 << "added an element with indices " << Indices
221 << " and frequency permutation " << permutations4[1] << " (" << pElement
222 << ").");
223 }
224 }
225 if(!SameCIndices && !SameCXIndices) {
226 IndexCombination4 Indices2143(Indices.Index2, Indices.Index1, Indices.Index4, Indices.Index3);
227 if(!isInContainer(Indices2143)) {
228 ElementsMap.emplace(Indices2143, ElementWithPermFreq<ElementType>(pElement, permutations4[7]));
229 DEBUG("IndexContainer4::fill() at " << this << ": "
230 << "added an element with indices " << Indices
231 << " and frequency permutation " << permutations4[7] << " (" << pElement
232 << ").");
233 }
234 }
235
236 return iter->second;

240inline ElementWithPermFreq<ElementType>&
241IndexContainer4<ElementType, SourceObject>::operator()(IndexCombination4 const& Indices) {
242 auto iter = ElementsMap.find(Indices);
243
244 if(iter == ElementsMap.end()) {
245 DEBUG("IndexContainer4 at " << this << ": "

247 << ", Index3=" << Indices.Index3 << ", Index4=" << Indices.Index4

249 return create(Indices);

262template <typename ElementType, typename SourceObject>
263inline std::set<IndexCombination4> IndexContainer4<ElementType, SourceObject>::enumerateIndices() const {
264 std::set<IndexCombination4> AllIndices;
265
266 for(ParticleIndex Index1 = 0; Index1 < NumIndices; ++Index1)
267 for(ParticleIndex Index2 = Index1; Index2 < NumIndices; ++Index2)
268 for(ParticleIndex Index3 = 0; Index3 < NumIndices; ++Index3)
269 for(ParticleIndex Index4 = Index3; Index4 < NumIndices; ++Index4)
270 AllIndices.emplace(Index1, Index2, Index3, Index4);
271
272 return AllIndices;
273}
