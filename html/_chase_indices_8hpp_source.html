29template <bool Complex>
30inline bool chaseIndices(typename RowMajorMatrixType<Complex>::InnerIterator& index1_iter,
31 typename ColMajorMatrixType<Complex>::InnerIterator& index2_iter) {
32 InnerQuantumState index1 = index1_iter.index();
33 InnerQuantumState index2 = index2_iter.index();
34
35 if(index1 == index2)
36 return true;
37
38 if(index1 < index2)
39 for(; InnerQuantumState(index1_iter.index()) < index2 && index1_iter; ++index1_iter)
40 ;
41 else
42 for(; InnerQuantumState(index2_iter.index()) < index1 && index2_iter; ++index2_iter)
43 ;
44
45 return false;
46}
