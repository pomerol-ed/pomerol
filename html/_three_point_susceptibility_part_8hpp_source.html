527inline ComplexType ThreePointSusceptibilityPart::NonResonantFFTerm::operator()(ComplexType z1, ComplexType z2) const {
528 return Coeff / ((z1 - Poles[0]) * (z2 - Poles[1]));
529}

531inline ComplexType ThreePointSusceptibilityPart::NonResonantFBTerm::operator()(ComplexType z1, ComplexType z2) const {
532 return Coeff / ((z1 - P1) * (z1 - double(xi) * z2 - P12));
533}

536ThreePointSusceptibilityPart::ResonantTerm::operator()(ComplexType z1, ComplexType z2, RealType DeltaTolerance) const {
