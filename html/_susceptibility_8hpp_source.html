141 inline ComplexType Susceptibility::operator()(ComplexType z) const {
142  ComplexType Value = 0;
143  if(!Vanishing) {
144  Value = std::accumulate(parts.begin(),
145  parts.end(),
146  ComplexType(0),
147  [z](ComplexType s, SusceptibilityPart const& p) { return s + p(z); });
148  }
149  if(SubtractDisconnected && std::abs(z) < 1e-15)
150  Value -= ave_A * ave_B * beta; // only for n=0
151  return Value;
152 }

154 inline ComplexType Susceptibility::of_tau(RealType tau) const {

162  if(SubtractDisconnected)
