147inline ComplexType Susceptibility::operator()(ComplexType z) const {
148 ComplexType Value = 0;
149 if(!Vanishing) {
150 Value = std::accumulate(parts.begin(),
151 parts.end(),
152 ComplexType(0),
153 [z](ComplexType s, SusceptibilityPart const& p) { return s + p(z); });
154 }
155 if(SubtractDisconnected && std::abs(z) <= std::numeric_limits<double>::epsilon())
156 Value -= ave_A * ave_B * beta; // only for n=0
157 return Value;
158}

160inline ComplexType Susceptibility::of_tau(RealType tau) const {
161 ComplexType Value = 0;
162 if(!Vanishing) {
163 Value = std::accumulate(parts.begin(),
164 parts.end(),
165 ComplexType(0),
166 [tau](ComplexType s, SusceptibilityPart const& p) { return s + p.of_tau(tau); });
167 }
168 if(SubtractDisconnected)
169 Value -= ave_A * ave_B;
170 return Value;
171}
